## L1-7 分寝室
```cpp
#include<bits/stdc++.h>
using namespace std;

int main()
{
    cin.tie(NULL)->sync_with_stdio(false);
    long long n0,n1,n;
    cin>>n0>>n1>>n;

    int min1=-1,min2=-1;
    long long min_diff=LLONG_MAX;

    for(int i=1;i<n;i++)
    {
        int room1=i;
        int room2=n-i;
        if(n0%room1==0&&n1%room2==0)
        {
            long long person1=n0/room1;
            long long person2=n1/room2;

            if(person1>=2&&person2>=2)
            {
                long long current_diff=abs(person1-person2);
                if(current_diff<min_diff)
                {
                    min_diff=current_diff;
                    min1=room1;
                    min2=room2;
                }
            }
        }
    }

    if(min1==-1)
    {
        cout<<"No Solution\n";
    }
    else
    {
        cout<<min1<<" "<<min2<<"\n";
    }

    return 0;
}
```

## L2-2 天梯赛的赛场安排
```cpp
#include<bits/stdc++.h>
using namespace std;
using ll=long long;

struct sl{
    string name;
    int val;
    int id;
    int ans=0;
};
struct rm{
    int free;
};
struct cmp{
    bool operator()(sl* a,sl* b)//大根判断重载
    {
        if(a->val!=b->val)return a->val<b->val;
        return a->id>b->id;
    }
};

int main()
{
    cin.tie(NULL)->sync_with_stdio(false);
    int n,c;
    cin>>n>>c;
    vector<sl> school(n);
    priority_queue<sl*,vector<sl*>,cmp> pq;
    for(int i=0;i<n;i++)
    {
        cin>>school[i].name>>school[i].val;
        school[i].id=i;
        pq.push(&school[i]);
    }
    vector<rm> room;
    while(!pq.empty())
    {
        sl* s=pq.top();
        pq.pop();
        if(s->val<=0)break;
        if(s->val>=c)
        {
            s->val-=c;
            s->ans++;
            rm temp;
            temp.free=0;
            room.push_back(temp);
            if(s->val>0)pq.push(s);
        }
        else
        {
            int found=-1;
            for(int j=0;j<(int)room.size();j++)
            {
                if(room[j].free>=s->val)
                {
                    found=j;
                    break;
                }
            }
            if(found!=-1)
            {
                room[found].free-=s->val;
            }
            else
            {
                rm temp;
                temp.free=c-s->val;
                room.push_back(temp);
            }
            s->val=0;
            s->ans++;
        }
    }
    for(int i=0;i<n;i++)
    {
        cout<<school[i].name<<' '<<school[i].ans<<"\n";
    }
    cout<<room.size()<<"\n";
    return 0;
}
```
